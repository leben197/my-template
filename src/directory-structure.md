# 项目目录结构

```
src/
├── api/                  # API接口封装
├── assets/               # 静态资源
│   ├── images/           # 图片资源
│   ├── fonts/            # 字体资源
│   └── style/            # 样式资源
├── components/           # 公共组件
│   ├── basic/            # 基础组件
│   └── business/         # 业务组件
├── composables/          # 组合式函数
├── config/               # 全局配置
├── constants/            # 常量定义
├── directives/           # 自定义指令
├── enums/                # 枚举定义
├── hooks/                # 自定义钩子
├── layouts/              # 布局组件
├── pages/                # 页面组件
├── router/               # 路由配置
├── stores/               # 状态管理
├── types/                # 类型定义
├── utils/                # 工具函数
│   ├── http.ts           # HTTP请求封装
│   ├── storage.ts        # 存储工具
│   └── tools.ts          # 通用工具函数
└── App.vue               # 根组件
```
